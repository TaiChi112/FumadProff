# Data Mining Demo

```python
import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import MinMaxScaler
from sklearn.tree import DecisionTreeClassifier
from sklearn.metrics import accuracy_score

# ==========================================
# 1. INPUT DATA (Data Collection)
# ==========================================
# ‡∏à‡∏≥‡∏•‡∏≠‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏•‡∏π‡∏Å‡∏Ñ‡πâ‡∏≤ 10 ‡∏Ñ‡∏ô (‡πÉ‡∏ô‡∏Ñ‡∏ß‡∏≤‡∏°‡πÄ‡∏õ‡πá‡∏ô‡∏à‡∏£‡∏¥‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏à‡∏∞‡∏°‡∏≤‡∏à‡∏≤‡∏Å SQL Database ‡∏´‡∏£‡∏∑‡∏≠ CSV)
# ‡∏™‡∏°‡∏°‡∏ï‡∏¥ 1 = ‡πÄ‡∏•‡∏¥‡∏Å‡πÄ‡∏õ‡πá‡∏ô‡∏™‡∏°‡∏≤‡∏ä‡∏¥‡∏Å (Churn), 0 = ‡∏¢‡∏±‡∏á‡∏≠‡∏¢‡∏π‡πà‡∏ï‡πà‡∏≠
data = {
    'Customer_ID': ['C001', 'C002', 'C003', 'C004', 'C005', 'C006', 'C007', 'C008', 'C009', 'C010'],
    'Name': ['Alice', 'Bob', 'Charlie', 'David', 'Eve', 'Frank', 'Grace', 'Hannah', 'Ian', 'Jack'],
    'Age': [25, 45, 35, 50, 23, 60, 32, 28, 40, 55],
    'Salary': [18000, 85000, 45000, 120000, 15000, 90000, 50000, 22000, 60000, 110000],
    'Days_Since_Last_Buy': [2, 45, 10, 5, 60, 3, 15, 40, 7, 2],
    'Total_Spend': [500, 12000, 3000, 25000, 200, 15000, 4000, 800, 5500, 30000],
    'Churn_Label': [0, 1, 0, 0, 1, 0, 0, 1, 0, 0] 
}

df = pd.DataFrame(data)

print("--- ‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏î‡∏¥‡∏ö (Raw Data) ---")
print(df.head(3)) # ‡∏î‡∏π‡πÅ‡∏Ñ‡πà 3 ‡πÅ‡∏ñ‡∏ß‡πÅ‡∏£‡∏Å
print("\n")


# ==========================================
# 2. DATA PREPROCESSING
# ==========================================

# --- 2.1 Feature Selection (‡πÄ‡∏•‡∏∑‡∏≠‡∏Å‡πÄ‡∏â‡∏û‡∏≤‡∏∞‡∏Ñ‡∏≠‡∏•‡∏±‡∏°‡∏ô‡πå‡∏ó‡∏µ‡πà‡∏°‡∏µ‡∏ú‡∏•) ---
# HINT: ID ‡∏Å‡∏±‡∏ö ‡∏ä‡∏∑‡πà‡∏≠ ‡πÄ‡∏õ‡πá‡∏ô Unique Identifier ‡πÑ‡∏°‡πà‡∏°‡∏µ‡∏ú‡∏•‡∏ï‡πà‡∏≠‡∏û‡∏§‡∏ï‡∏¥‡∏Å‡∏£‡∏£‡∏°‡∏Å‡∏≤‡∏£‡∏ã‡∏∑‡πâ‡∏≠ ‡∏ï‡∏±‡∏î‡∏ó‡∏¥‡πâ‡∏á‡πÑ‡∏î‡πâ‡πÄ‡∏•‡∏¢
# ‡∏ô‡∏µ‡πà‡∏Ñ‡∏∑‡∏≠‡∏Å‡∏≤‡∏£‡∏•‡∏î Dimensions ‡πÅ‡∏ö‡∏ö‡∏á‡πà‡∏≤‡∏¢‡∏ó‡∏µ‡πà‡∏™‡∏∏‡∏î (Manual Feature Selection)
features = df.drop(['Customer_ID', 'Name', 'Churn_Label'], axis=1)
labels = df['Churn_Label']

# --- 2.2 Data Subsetting (Train/Test Split) ---
# HINT: ‡πÅ‡∏ö‡πà‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡πÑ‡∏ß‡πâ "‡∏™‡∏≠‡∏ô" (Train) ‡πÅ‡∏•‡∏∞ "‡∏™‡∏≠‡∏ö" (Test) ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏à‡∏≥‡∏•‡∏≠‡∏á‡∏Å‡∏≤‡∏£‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô‡∏à‡∏£‡∏¥‡∏á
# ‡∏™‡∏∏‡πà‡∏° 80% ‡πÑ‡∏õ‡∏™‡∏≠‡∏ô, 20% ‡πÑ‡∏õ‡∏™‡∏≠‡∏ö
X_train, X_test, y_train, y_test = train_test_split(features, labels, test_size=0.2, random_state=42)

# --- 2.3 Normalization (‡∏õ‡∏£‡∏±‡∏ö‡∏™‡πÄ‡∏Å‡∏•‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•) ---
# HINT: Salary ‡∏´‡∏•‡∏±‡∏Å‡∏´‡∏°‡∏∑‡πà‡∏ô vs Age ‡∏´‡∏•‡∏±‡∏Å‡∏™‡∏¥‡∏ö ‡∏ñ‡πâ‡∏≤‡πÑ‡∏°‡πà‡∏õ‡∏£‡∏±‡∏ö ‡πÇ‡∏°‡πÄ‡∏î‡∏•‡∏à‡∏∞‡∏•‡∏≥‡πÄ‡∏≠‡∏µ‡∏¢‡∏á‡πÑ‡∏õ‡∏ó‡∏≤‡∏á Salary
# ‡πÄ‡∏£‡∏≤‡πÉ‡∏ä‡πâ Min-Max Scaler ‡∏õ‡∏£‡∏±‡∏ö‡∏ó‡∏∏‡∏Å‡∏≠‡∏¢‡πà‡∏≤‡∏á‡πÉ‡∏´‡πâ‡∏≠‡∏¢‡∏π‡πà‡πÉ‡∏ô‡∏ä‡πà‡∏ß‡∏á 0 ‡∏ñ‡∏∂‡∏á 1
scaler = MinMaxScaler()

# fit_transform ‡∏Å‡∏±‡∏ö‡∏ï‡∏±‡∏ß Train (‡∏à‡∏≥‡∏Ñ‡πà‡∏≤ Min/Max ‡∏Ç‡∏≠‡∏á Train ‡πÑ‡∏ß‡πâ)
X_train_scaled = scaler.fit_transform(X_train) 
# transform ‡∏ï‡∏±‡∏ß Test (‡πÉ‡∏ä‡πâ‡∏Ñ‡πà‡∏≤ Min/Max ‡πÄ‡∏î‡∏¥‡∏°‡∏à‡∏≤‡∏Å Train ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏Ñ‡∏ß‡∏≤‡∏°‡πÅ‡∏ü‡∏£‡πå)
X_test_scaled = scaler.transform(X_test)

print("--- ‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏´‡∏•‡∏±‡∏á Preprocessing (Normalized) ---")
print(pd.DataFrame(X_train_scaled, columns=features.columns).head(3))
print("\n")


# ==========================================
# 3. DATA MINING
# ==========================================

# ‡πÄ‡∏•‡∏∑‡∏≠‡∏Å Algorithm: Decision Tree (‡∏ï‡πâ‡∏ô‡πÑ‡∏°‡πâ‡∏ï‡∏±‡∏î‡∏™‡∏¥‡∏ô‡πÉ‡∏à)
# ‡πÄ‡∏û‡∏£‡∏≤‡∏∞‡πÄ‡∏Ç‡πâ‡∏≤‡πÉ‡∏à‡∏á‡πà‡∏≤‡∏¢ ‡πÄ‡∏´‡∏°‡∏≤‡∏∞‡∏Å‡∏±‡∏ö‡∏Å‡∏≤‡∏£‡πÄ‡∏£‡∏¥‡πà‡∏°‡πÄ‡∏£‡∏µ‡∏¢‡∏ô‡∏£‡∏π‡πâ
clf = DecisionTreeClassifier(random_state=42)

# ‡∏™‡∏±‡πà‡∏á‡πÉ‡∏´‡πâ‡πÇ‡∏°‡πÄ‡∏î‡∏• "‡πÄ‡∏£‡∏µ‡∏¢‡∏ô‡∏£‡∏π‡πâ" ‡∏à‡∏≤‡∏Å‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏ó‡∏µ‡πà‡πÄ‡∏ï‡∏£‡∏µ‡∏¢‡∏°‡πÑ‡∏ß‡πâ (Machine Learning)
clf.fit(X_train_scaled, y_train)

# ‡∏•‡∏≠‡∏á‡∏ó‡∏≥‡∏ô‡∏≤‡∏¢‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏ä‡∏∏‡∏î‡∏™‡∏≠‡∏ö (Test Set)
y_pred = clf.predict(X_test_scaled)


# ==========================================
# 4. POSTPROCESSING & INFORMATION
# ==========================================

# --- 4.1 Evaluation (‡∏ß‡∏±‡∏î‡∏ú‡∏•) ---
accuracy = accuracy_score(y_test, y_pred)
print(f"--- ‡∏ú‡∏•‡∏•‡∏±‡∏û‡∏ò‡πå (Information) ---")
print(f"‡∏Ñ‡∏ß‡∏≤‡∏°‡πÅ‡∏°‡πà‡∏ô‡∏¢‡∏≥‡∏Ç‡∏≠‡∏á‡πÇ‡∏°‡πÄ‡∏î‡∏•: {accuracy * 100:.2f}%")

# --- 4.2 Pattern Interpretation (‡∏î‡∏π‡∏ß‡πà‡∏≤‡πÇ‡∏°‡πÄ‡∏î‡∏•‡∏£‡∏π‡πâ‡∏≠‡∏∞‡πÑ‡∏£) ---
# HINT: ‡∏î‡∏π‡∏ß‡πà‡∏≤ Feature ‡πÑ‡∏´‡∏ô‡∏™‡∏≥‡∏Ñ‡∏±‡∏ç‡∏ó‡∏µ‡πà‡∏™‡∏∏‡∏î‡πÉ‡∏ô‡∏Å‡∏≤‡∏£‡∏ï‡∏±‡∏î‡∏™‡∏¥‡∏ô‡πÉ‡∏à‡∏ß‡πà‡∏≤‡∏•‡∏π‡∏Å‡∏Ñ‡πâ‡∏≤‡∏à‡∏∞‡πÄ‡∏•‡∏¥‡∏Å‡∏´‡∏£‡∏∑‡∏≠‡πÑ‡∏°‡πà‡πÄ‡∏•‡∏¥‡∏Å
feature_importance = pd.DataFrame({
    'Feature': features.columns,
    'Importance': clf.feature_importances_
}).sort_values(by='Importance', ascending=False)

print("\n‡∏õ‡∏±‡∏à‡∏à‡∏±‡∏¢‡∏ó‡∏µ‡πà‡∏°‡∏µ‡∏ú‡∏•‡∏ï‡πà‡∏≠‡∏Å‡∏≤‡∏£‡πÄ‡∏•‡∏¥‡∏Å‡πÄ‡∏õ‡πá‡∏ô‡∏™‡∏°‡∏≤‡∏ä‡∏¥‡∏Å (‡πÄ‡∏£‡∏µ‡∏¢‡∏á‡∏ï‡∏≤‡∏°‡∏Ñ‡∏ß‡∏≤‡∏°‡∏™‡∏≥‡∏Ñ‡∏±‡∏ç):")
print(feature_importance)
```

## ‡∏Ñ‡∏≥‡∏≠‡∏ò‡∏¥‡∏ö‡∏≤‡∏¢‡πÄ‡∏û‡∏¥‡πà‡∏°‡πÄ‡∏ï‡∏¥‡∏°

### 1. Data Preprocessing {/*(‡∏à‡∏∏‡∏î‡∏ó‡∏µ‡πà‡∏Ñ‡∏∏‡∏ì‡∏ñ‡∏≤‡∏°‡∏ñ‡∏∂‡∏á) */}

‡πÉ‡∏ô Code ‡∏ô‡∏µ‡πâ‡∏ú‡∏°‡∏£‡∏ß‡∏°‡∏Ç‡∏±‡πâ‡∏ô‡∏ï‡∏≠‡∏ô‡∏¢‡πà‡∏≠‡∏¢‡πÜ ‡πÑ‡∏ß‡πâ‡πÉ‡∏ô‡∏ä‡πà‡∏ß‡∏á Step 2:

- **Feature Selection**: ‡∏Ñ‡∏≥‡∏™‡∏±‡πà‡∏á `df.drop(['Customer_ID', 'Name'...])` ‡∏Ñ‡∏∑‡∏≠‡∏Å‡∏≤‡∏£‡∏ö‡∏≠‡∏Å "‡∏≠‡∏¢‡πà‡∏≤‡πÄ‡∏≠‡∏≤ Name ‡∏Ñ‡∏ô‡∏°‡∏≤‡∏Ñ‡∏¥‡∏î‡∏ô‡∏∞ ‡∏°‡∏±‡∏ô‡πÑ‡∏°‡πà‡πÄ‡∏Å‡∏µ‡πà‡∏¢‡∏ß" ‡∏ô‡∏µ‡πà‡∏Ñ‡∏∑‡∏≠‡∏Å‡∏≤‡∏£‡∏•‡∏î Noise ‡∏Ñ‡∏£‡∏±‡∏ö

- **Dimensionality Reduction**: ‡πÉ‡∏ô Code ‡∏ô‡∏µ‡πâ‡πÄ‡∏£‡∏≤‡∏ó‡∏≥‡πÅ‡∏ö‡∏ö Basic ‡∏Ñ‡∏∑‡∏≠‡∏Å‡∏≤‡∏£ "‡∏ï‡∏±‡∏î column ‡∏ó‡∏¥‡πâ‡∏á" ‡πÅ‡∏ï‡πà‡∏ñ‡πâ‡∏≤‡πÄ‡∏õ‡πá‡∏ô‡∏Ç‡∏±‡πâ‡∏ô‡∏™‡∏π‡∏á (‡πÉ‡∏ô‡∏´‡∏ô‡∏±‡∏á‡∏™‡∏∑‡∏≠ Tan) ‡∏≠‡∏≤‡∏à‡∏à‡∏∞‡πÉ‡∏ä‡πâ‡∏Ñ‡∏≥‡∏™‡∏±‡πà‡∏á PCA ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏¢‡∏∏‡∏ö‡∏£‡∏ß‡∏° `Salary + Total_Spend` ‡πÉ‡∏´‡πâ‡πÄ‡∏´‡∏•‡∏∑‡∏≠‡∏ï‡∏±‡∏ß‡πÅ‡∏õ‡∏£‡πÄ‡∏î‡∏µ‡∏¢‡∏ß ‡πÅ‡∏ï‡πà‡πÉ‡∏ô‡∏ó‡∏≤‡∏á‡∏õ‡∏è‡∏¥‡∏ö‡∏±‡∏ï‡∏¥‡πÄ‡∏ö‡∏∑‡πâ‡∏≠‡∏á‡∏ï‡πâ‡∏ô ‡∏Å‡∏≤‡∏£‡πÄ‡∏•‡∏∑‡∏≠‡∏Å Feature ‡∏ó‡∏µ‡πà‡∏î‡∏µ‡∏Å‡πá‡πÄ‡∏û‡∏µ‡∏¢‡∏á‡∏û‡∏≠‡πÅ‡∏•‡πâ‡∏ß‡∏Ñ‡∏£‡∏±‡∏ö

- **Normalization**: ‡∏Ñ‡∏≥‡∏™‡∏±‡πà‡∏á `MinMaxScaler` ‡∏™‡∏≥‡∏Ñ‡∏±‡∏ç‡∏°‡∏≤‡∏Å ‡∏™‡∏±‡∏á‡πÄ‡∏Å‡∏ï‡∏ß‡πà‡∏≤ Salary (‡∏´‡∏•‡∏±‡∏Å‡πÅ‡∏™‡∏ô) ‡∏à‡∏∞‡∏ñ‡∏π‡∏Å‡∏¢‡πà‡∏≠‡πÄ‡∏´‡∏•‡∏∑‡∏≠‡πÅ‡∏Ñ‡πà `0.x` ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡πÉ‡∏´‡πâ‡πÄ‡∏ó‡πà‡∏≤‡πÄ‡∏ó‡∏µ‡∏¢‡∏°‡∏Å‡∏±‡∏ö Age (‡∏´‡∏•‡∏±‡∏Å‡∏™‡∏¥‡∏ö) ‡∏ñ‡πâ‡∏≤‡πÑ‡∏°‡πà‡∏ó‡∏≥‡∏Ç‡∏±‡πâ‡∏ô‡∏ï‡∏≠‡∏ô‡∏ô‡∏µ‡πâ ‡πÇ‡∏°‡πÄ‡∏î‡∏•‡∏à‡∏∞‡∏°‡∏≠‡∏á‡∏ß‡πà‡∏≤ Salary ‡∏™‡∏≥‡∏Ñ‡∏±‡∏ç‡∏ó‡∏µ‡πà‡∏™‡∏∏‡∏î‡πÄ‡∏™‡∏°‡∏≠‡πÄ‡∏û‡∏£‡∏≤‡∏∞‡∏ï‡∏±‡∏ß‡πÄ‡∏•‡∏Ç‡∏°‡∏±‡∏ô‡πÄ‡∏¢‡∏≠‡∏∞

### 2. Data Mining vs Algorithm

- **Data Mining** ‡∏Ñ‡∏∑‡∏≠‡∏ä‡∏∑‡πà‡∏≠‡∏Å‡∏£‡∏∞‡∏ö‡∏ß‡∏ô‡∏Å‡∏≤‡∏£‡∏†‡∏≤‡∏û‡πÉ‡∏´‡∏ç‡πà
- **Decision Tree Classifier** ‡∏Ñ‡∏∑‡∏≠ "‡πÄ‡∏Ñ‡∏£‡∏∑‡πà‡∏≠‡∏á‡∏°‡∏∑‡∏≠" ‡∏´‡∏£‡∏∑‡∏≠ Algorithm ‡∏ó‡∏µ‡πà‡πÄ‡∏£‡∏≤‡∏´‡∏¢‡∏¥‡∏ö‡∏°‡∏≤‡πÉ‡∏ä‡πâ‡πÉ‡∏ô‡∏Ç‡∏±‡πâ‡∏ô‡∏ï‡∏≠‡∏ô‡∏ô‡∏µ‡πâ (‡πÄ‡∏´‡∏°‡∏∑‡∏≠‡∏ô‡πÄ‡∏£‡∏≤‡∏´‡∏¢‡∏¥‡∏ö‡∏™‡∏ß‡πà‡∏≤‡∏ô‡∏°‡∏≤‡πÄ‡∏à‡∏≤‡∏∞‡∏£‡∏π)

### 3. Postprocessing

- ‡∏ú‡∏•‡∏•‡∏±‡∏û‡∏ò‡πå‡∏ó‡∏µ‡πà‡πÄ‡∏õ‡πá‡∏ô‡∏ï‡∏±‡∏ß‡πÄ‡∏•‡∏Ç (`y_pred`) ‡∏≠‡∏≤‡∏à‡∏à‡∏∞‡∏î‡∏π‡∏¢‡∏≤‡∏Å
- ‡πÄ‡∏£‡∏≤‡∏à‡∏∂‡∏á‡∏ó‡∏≥ `feature_importance` ‡∏≠‡∏≠‡∏Å‡∏°‡∏≤ ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡πÅ‡∏õ‡∏•‡∏á‡∏Ñ‡∏ì‡∏¥‡∏ï‡∏®‡∏≤‡∏™‡∏ï‡∏£‡πå‡πÉ‡∏´‡πâ‡πÄ‡∏õ‡πá‡∏ô‡∏†‡∏≤‡∏©‡∏≤‡∏ò‡∏∏‡∏£‡∏Å‡∏¥‡∏à‡∏ß‡πà‡∏≤:
  
  > "‡∏≠‡πã‡∏≠... ‡∏™‡∏¥‡πà‡∏á‡∏™‡∏≥‡∏Ñ‡∏±‡∏ç‡∏ó‡∏µ‡πà‡∏™‡∏∏‡∏î‡∏Ñ‡∏∑‡∏≠ ‡∏à‡∏≥‡∏ô‡∏ß‡∏ô‡∏ß‡∏±‡∏ô‡∏ó‡∏µ‡πà‡∏´‡∏≤‡∏¢‡πÑ‡∏õ (Days_Since_Last_Buy) ‡∏ñ‡πâ‡∏≤‡∏´‡∏≤‡∏¢‡πÑ‡∏õ‡∏ô‡∏≤‡∏ô ‡∏°‡∏µ‡πÇ‡∏≠‡∏Å‡∏≤‡∏™‡πÄ‡∏•‡∏¥‡∏Å‡∏™‡∏π‡∏á"
  
  **‡∏ô‡∏µ‡πà‡πÅ‡∏´‡∏•‡∏∞‡∏Ñ‡∏£‡∏±‡∏ö‡∏Ñ‡∏∑‡∏≠ Knowledge ‡∏ó‡∏µ‡πà‡∏Ñ‡πâ‡∏ô‡∏û‡∏ö**

```python
# ==========================================
# 5. VISUALIZATION (‡πÅ‡∏ñ‡∏°‡∏û‡∏¥‡πÄ‡∏®‡∏©: ‡∏î‡∏π‡∏´‡∏ô‡πâ‡∏≤‡∏ï‡∏≤‡πÇ‡∏°‡πÄ‡∏î‡∏•)
# ==========================================
import matplotlib.pyplot as plt
from sklearn.tree import plot_tree

# ‡∏Å‡∏≥‡∏´‡∏ô‡∏î‡∏Ç‡∏ô‡∏≤‡∏î‡∏Ç‡∏≠‡∏á‡∏†‡∏≤‡∏û‡∏ó‡∏µ‡πà‡∏à‡∏∞‡∏ß‡∏≤‡∏î (‡∏Å‡∏ß‡πâ‡∏≤‡∏á 15 ‡∏ô‡∏¥‡πâ‡∏ß, ‡∏™‡∏π‡∏á 10 ‡∏ô‡∏¥‡πâ‡∏ß) agar ‡πÉ‡∏´‡πâ‡πÄ‡∏´‡πá‡∏ô‡∏ä‡∏±‡∏î‡πÜ

plot_tree(clf, 
          feature_names=features.columns,  # ‡πÉ‡∏™‡πà‡∏ä‡∏∑‡πà‡∏≠‡∏Ñ‡∏≠‡∏•‡∏±‡∏°‡∏ô‡πå‡πÄ‡∏û‡∏∑‡πà‡∏≠‡πÉ‡∏´‡πâ‡∏£‡πà‡∏≠‡∏á‡∏£‡∏π‡πâ‡∏á‡πà‡∏≤‡∏¢‡∏Ç‡∏∂‡πâ‡∏ô (Age, Salary)
          class_names=['Stay (0)', 'Churn (1)'], # ‡∏ï‡∏±‡πâ‡∏á‡∏ä‡∏∑‡πà‡∏≠‡πÉ‡∏´‡πâ‡∏ú‡∏•‡∏•‡∏±‡∏û‡∏ò‡πå (0=‡∏≠‡∏¢‡∏π‡πà‡∏ï‡πà‡∏≠, 1=‡πÄ‡∏•‡∏¥‡∏Å)
          filled=True,     # ‡∏£‡∏∞‡∏ö‡∏≤‡∏¢‡∏™‡∏µ‡∏Å‡∏•‡πà‡∏≠‡∏á‡∏ï‡∏≤‡∏°‡∏Å‡∏•‡∏∏‡πà‡∏°‡∏ú‡∏•‡∏•‡∏±‡∏û‡∏ò‡πå (‡∏™‡∏µ‡∏ü‡πâ‡∏≤=‡∏≠‡∏¢‡∏π‡πà‡∏ï‡πà‡∏≠, ‡∏™‡∏µ‡∏™‡πâ‡∏°=‡πÄ‡∏•‡∏¥‡∏Å)
          rounded=True,    # ‡∏ó‡∏≥‡∏Ç‡∏≠‡∏ö‡∏°‡∏ô‡πÜ ‡πÉ‡∏´‡πâ‡∏™‡∏ß‡∏¢‡∏á‡∏≤‡∏°
          fontsize=12)     # ‡∏Ç‡∏ô‡∏≤‡∏î‡∏ï‡∏±‡∏ß‡∏≠‡∏±‡∏Å‡∏©‡∏£

plt.title("Decision Tree Logic for Customer Churn Prediction")
plt.show() # ‡πÅ‡∏™‡∏î‡∏á‡∏†‡∏≤‡∏û‡∏Ç‡∏∂‡πâ‡∏ô‡∏°‡∏≤
```
